#include<stdio.h>
int main()
{
    char s1[101],s2[101],s3[105];
    int n,i,l1,l2,large,small,carry,j,countnum,finallen;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%s",s1);
        scanf("%s",s2);
        l1=strlen(s1);
        l2=strlen(s2);
        if(l1>=l2)
            countnum=1;
        else
            countnum=2;
        large=(l1>=l2)?l1:l2;
        small=(l1<=l2)?l1:l2;
        carry=0;
        for(j=1;j<=small;j++)
        {
            if((s1[l1-j]=='1')&&(s2[l2-j]=='1'))
            {
                if(carry==1)
                {
                    s3[j-1]='1';
                    carry=1;
                }
                else
                {
                    s3[j-1]='0';
                    carry=1;
                }
            }
            else if((s1[l1-j]=='1')||(s2[l2-j]=='1'))
            {
                if(carry==1)
                {
                    s3[j-1]='0';
                    carry=1;
                }
                else
                {
                    s3[j-1]='1';
                    carry=0;
                }
            }
            else
            {
                if(carry==1)
                {
                    s3[j-1]='1';
                    carry=0;
                }
                else
                {
                    s3[j-1]='0';
                    carry=0;
                }
            }
        }
        if(countnum==1)
        {
            for(j;j<=large;j++)
            {
                if((s1[large-j]=='1')&&(carry==1))
                {
                    s3[j-1]='0';
                    carry=1;
                }
                else if ((s1[large-j]=='1')||(carry==1))
                {
                    s3[j-1]='1';
                    carry=0;
                }
                else
                {
                    s3[j-1]='0';
                    carry=0;
                }
            }
        }
        else
        {
            for(j;j<=large;j++)
            {
                if((s2[large-j]=='1')&&(carry==1))
                {
                    s3[j-1]='0';
                    carry=1;
                }
                else if((s2[large-j]=='1')||(carry==1))
                {
                    s3[j-1]='1';
                    carry=0;
                }
                else
                {
                    s3[j-1]='0';
                    carry=0;
                }
            }
        }
    if(carry==1){
        s3[j-1]='1';
    j=j+1;}
    s3[j]='/0';
    finallen=strlen(s3);
    for(j=finallen-1;j>=0;j--)
        printf("%c",s3[j]);
    printf("\n");
    }
    return 0;
}
